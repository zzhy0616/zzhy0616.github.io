深度优先搜索DFS

简单来说就是一条路莽到底

部分和问题

int a[MAX_n];
int n, k;

// 已经从前i项得到了和sum， 然后对于i项之后的进行分支
bool dfs(int i, int sum){
    //如果前i项都计算过了则返回sum是否与k相等
    if(i == n) return sum == k;
    
    //不加上a[i]的情况
    if(dfs(i + 1, sum)) return true;

    //加上a[i]的情况
    if(dfs(i + 1, sum + a[i])) return true;

    //无论是否加上a[i]都不能凑成k就返回false
    return false;
}

void solve(){
    if(dfs(0,0)) printf("Yes\n");
    else printf("No\n");
}
